######Ruby

# .lenght - retorna o tamanho da variavel
nome_da_variavel.lenght

nome_da_variavel.reverse

#retorna o conteudo da variavel em letras minusculas
nome_da_variavel.downcase

#metodos encadeados
nome_da_variavel.upcase.reverse.downcase
